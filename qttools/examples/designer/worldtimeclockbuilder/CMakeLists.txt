)

    WIN32_EXECUTABLE TRUE
    MACOSX_BUNDLE TRUE
)

    Qt::Core

    Qt::Widgets
)

    "form.ui"

)

    RUNTIME DESTINATION "${INSTALL_EXAMPLEDIR}"
